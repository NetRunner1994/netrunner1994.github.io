function pickUseCase(el, type) {
  document.querySelectorAll('.uc-card').forEach(c=>c.classList.remove('active'));
  el.classList.add('active');
  S.useCase = type;
  document.getElementById('hid-usecase').value = type;
  setTimeout(()=>document.getElementById('builder').scrollIntoView({behavior:'smooth',block:'start'}),320);
}

function pickBudget(el, tier, lo, hi) {
  document.querySelectorAll('.bud-btn').forEach(b=>b.classList.remove('active'));
  el.classList.add('active');
  S.budget = tier;
  // Reset tier-specific selections
  ['gpu','cpu','mobo','ram','psu','cpu_cooler'].forEach(k=>{
    S[k]={name:null,lo:0,hi:0};
    const pp = document.getElementById('pp-'+ppId(k));
    if(pp){pp.className='pval empty';pp.textContent='—';}
  });
  const d = DATA[tier];
  buildOpts('gpu-opts', d.gpu, 'gpu');
  buildOpts('cpu-opts', d.cpu, 'cpu');
  buildOpts('mobo-opts', d.mobo, 'mobo');
  buildOpts('ram-opts', d.ram, 'ram');
  buildOpts('psu-opts', d.psu, 'psu');
  buildOpts('cooler-opts', d.cpu_cooler, 'cpu_cooler');
  updateTotal();
  updateProgress();
  checkCompatibility();
}

function ppId(k){
  const m={gpu:'gpu',cpu:'cpu',mobo:'mobo',ram:'ram',storage:'storage',case:'case',psu:'psu',cpu_cooler:'cooler',case_fans:'fans'};
  return m[k]||k;
}

function buildOpts(containerId, items, type) {
  const c = document.getElementById(containerId);
  if(!c) return;
  c.innerHTML='';
  items.forEach(item=>{
    const b=document.createElement('button');
    b.className='opt-btn';
    b.innerHTML=`<span class="opt-name">${item.name}</span><span class="opt-price">~$${item.lo}–${item.hi}</span><span class="opt-spec">${item.spec}</span>`;
    b.onclick=function(){pickOpt(this,type,item.name,item.lo,item.hi);};
    c.appendChild(b);
  });
}

function pickOpt(el, type, name, lo, hi) {
  const container = el.closest('.copts') || el.closest('.storage-grp-body');
  if(container) container.querySelectorAll('.opt-btn').forEach(b=>b.classList.remove('active'));
  el.classList.add('active');
  S[type]={name,lo,hi};
  const ppEl=document.getElementById('pp-'+ppId(type));
  if(ppEl){ppEl.className='pval';ppEl.textContent=`~$${lo}–${hi}`;}
  updateTotal();
  updateProgress();
  updateSummary();
  if(type==='cpu'||type==='mobo') checkCompatibility();
  if(type==='ram') checkRamCompat();
}

function updateTotal() {
  const parts=['gpu','cpu','mobo','ram','storage','case','psu','cpu_cooler','case_fans'];
  let tLo=0,tHi=0;
  parts.forEach(p=>{if(S[p]&&S[p].lo>0){tLo+=S[p].lo;tHi+=S[p].hi;}});
  const el=document.getElementById('pp-total');
  if(tLo>0){
    animateTotal(el,prevTotalLo,tLo,tHi);
    prevTotalLo=tLo;
  }
}

function animateTotal(el,from,toMin,toMax) {
  const dur=380,t0=performance.now();
  const tick=t=>{
    const p=Math.min((t-t0)/dur,1);
    const e=1-Math.pow(1-p,3);
    const v=Math.round(from+(toMin-from)*e);
    el.textContent=`$${v.toLocaleString()}–$${toMax.toLocaleString()}`;
    if(p<1)requestAnimationFrame(tick);
    else el.textContent=`$${toMin.toLocaleString()}–$${toMax.toLocaleString()}`;
  };
  requestAnimationFrame(tick);
}

function updateProgress() {
  const required=['gpu','cpu','mobo','ram','storage','case','psu','cpu_cooler','case_fans'];
  const sel=required.filter(k=>S[k]&&S[k].name).length;
  const pct=Math.round((sel/required.length)*100);
  document.getElementById('prog-bar').style.width=pct+'%';
  document.getElementById('prog-text').textContent=`${sel} of ${required.length} selected`;
  document.getElementById('pp-prog-bar').style.width=pct+'%';
  document.getElementById('pp-prog-text').textContent=`${sel}/9`;
}

function checkCompatibility() {
  const box=document.getElementById('compat-box');
  const txt=document.getElementById('compat-text');
  if(!box||!txt) return;
  if(!S.cpu.name&&!S.mobo.name){
    box.className='compat-box neutral';
    txt.textContent=currentLang==='es'?'Selecciona CPU y Tarjeta Madre para verificar compatibilidad':'Select CPU & Motherboard to check compatibility';
    return;
  }
  if(!S.cpu.name||!S.mobo.name){
    box.className='compat-box neutral';
    txt.textContent=currentLang==='es'?'Selecciona ambos CPU y Tarjeta Madre para verificar':'Select both CPU and Motherboard to verify';
    return;
  }
  const cpuSocket=SOCKET[S.cpu.name];
  const moboSocket=SOCKET[S.mobo.name];
  if(!cpuSocket||!moboSocket){
    box.className='compat-box neutral';
    txt.textContent='Socket data unavailable — reach out to verify';
    return;
  }
  if(cpuSocket===moboSocket){
    box.className='compat-box ok';
    txt.textContent=`✓ Compatible — ${cpuSocket} socket match`;
  } else {
    box.className='compat-box error';
    txt.textContent=`⚠ Mismatch! CPU needs ${cpuSocket} — board is ${moboSocket}`;
  }
  checkRamCompat();
}

function checkRamCompat() {
  if(!S.cpu.name||!S.ram.name) return;
  const cpuSocket=SOCKET[S.cpu.name];
  if(!cpuSocket) return;
  const expectedDdr=DDR_MAP[cpuSocket];
  const ramName=S.ram.name.toLowerCase();
  const isDdr5=ramName.includes('ddr5');
  const isDdr4=ramName.includes('ddr4');
  const box=document.getElementById('compat-box');
  if(!isDdr4&&!isDdr5) return;
  if(expectedDdr==='DDR5'&&isDdr4){
    box.className='compat-box error';
    document.getElementById('compat-text').textContent=`⚠ RAM mismatch! ${cpuSocket} needs DDR5 — you selected DDR4`;
  } else if(expectedDdr==='DDR4'&&isDdr5){
    box.className='compat-box error';
    document.getElementById('compat-text').textContent=`⚠ RAM mismatch! ${cpuSocket} needs DDR4 — you selected DDR5`;
  }
}

function updateSummary() {
  const ucLabels={gaming:'Gaming',content:'Content Creation',ai:'AI & Productivity',general:'General Use'};
  const budLabels={budget:'Budget ($700–$1,100)',mid:'Mid-Range ($1,100–$1,800)',high:'High-End ($2,000–$3,200)',ultra:'Ultra ($4,500+)'};
  const rows=[
    {label:'Use Case',val:S.useCase?ucLabels[S.useCase]:null},
    {label:'Budget',val:S.budget?budLabels[S.budget]:null},
    {label:'GPU',val:S.gpu.name},
    {label:'CPU',val:S.cpu.name},
    {label:'Motherboard',val:S.mobo.name},
    {label:'RAM',val:S.ram.name},
    {label:'Storage',val:S.storage.name},
    {label:'Case',val:S.case.name},
    {label:'PSU',val:S.psu.name},
    {label:'CPU Cooler',val:S.cpu_cooler.name},
    {label:'Case Fans',val:S.case_fans.name}
  ].filter(r=>r.val);
  const wrap=document.getElementById('summary-lines');
  if(!rows.length){
    wrap.innerHTML='<div class="sline"><span style="color:var(--muted)">No components selected yet — configure your build above.</span></div>';
    document.getElementById('hid-summary').value='';
    return;
  }
  wrap.innerHTML=rows.map(r=>`<div class="sline"><span>${r.label}</span><span class="sval">${r.val}</span></div>`).join('');
  document.getElementById('hid-summary').value=rows.map(r=>`${r.label}: ${r.val}`).join(' | ');
}

function onSubmit(e) {
  e.preventDefault();
  updateSummary();
  const form=e.target;
  const body=new FormData(form);
  fetch('/',{method:'POST',body,headers:{'Accept':'application/json'}})
    .then(r=>{
      if(r.ok||r.status===200){
        document.getElementById('form-wrap').style.display='none';
        const sm=document.getElementById('success-msg');
        sm.style.display='block';
        sm.scrollIntoView({behavior:'smooth'});
      } else {
        // Fallback: just show success since Netlify may redirect
        document.getElementById('form-wrap').style.display='none';
        document.getElementById('success-msg').style.display='block';
      }
    })
    .catch(()=>{
      document.getElementById('form-wrap').style.display='none';
      document.getElementById('success-msg').style.display='block';
    });
}

  if(S.ram.name) params.set('ram',S.ram.name);
  if(S.storage.name) params.set('sto',S.storage.name);
  if(S.case.name) params.set('cas',S.case.name);
  if(S.psu.name) params.set('psu',S.psu.name);
  if(S.cpu_cooler.name) params.set('cool',S.cpu_cooler.name);
  if(S.case_fans.name) params.set('fans',S.case_fans.name);
  const url=window.location.origin+window.location.pathname+'?'+params.toString();
  document.getElementById('share-url').textContent=url;
  document.getElementById('share-url').style.display='block';
  document.getElementById('share-modal').classList.add('show');
}

function setLang(l) {
  currentLang=l;
  localStorage.setItem('ctlang',l);
  document.documentElement.lang=l;
  document.querySelectorAll('[data-en]').forEach(el=>{
    const val=l==='es'?el.dataset.es:el.dataset.en;
    if(val!==undefined) el.innerHTML=val;
  });
  document.querySelectorAll('[data-en-placeholder]').forEach(el=>{
    el.placeholder=l==='es'?el.dataset.esPlaceholder:el.dataset.enPlaceholder;
  });
  document.getElementById('lang-en').classList.toggle('active',l==='en');
  document.getElementById('lang-es').classList.toggle('active',l==='es');
}
